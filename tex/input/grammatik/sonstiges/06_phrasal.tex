\bigskip
\noindent\textbf{Wichtige ver- Verben:}

\begin{itemize}
    \item \textbf{verstehen} - to understand
    \item \textbf{verlieren} - to lose
    \item \textbf{versprechen} - to promise
    \item \textbf{verlassen} - to leave (a place)
    \item \textbf{verraten} - to betray, reveal
    \item \textbf{verbringen} - to spend (time)
    \item \textbf{versuchen} - to try
    \item \textbf{verhindern} - to prevent
    \item \textbf{verbessern} - to improve
    \item \textbf{vermitteln} - to mediate, teach
\end{itemize}

\bigskip
\noindent\textbf{Regel:}
\begin{itemize}
    \item \textbf{Trennbar}: betont,Prefix mit 1-2 Silben (ab-, an-, auf-, aus-, ein-, mit-, nach-, vor-, zu-, zurück-, durch-, über-, um-, unter-)
    \item \textbf{Untrennbar}: unbetont, Präfixe be-, emp-, ent-, er-, ge-, ver-, zer-
\end{itemize}

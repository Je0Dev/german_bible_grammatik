\section{Referenzwerke}
\begin{itemize}
    \item \textbf{\color{b1color}Grammatik aktiv B1+/B2/C1} - Jin/Voß - Cornelsen
    \item \textbf{\color{b1color}Lehr- und Übungsbuch der deutschen Grammatik} - Dreyer/Schmitt
    \item \textbf{\color{b1color}B-Grammatik} - Buscha/Szita - Klett
    \item \textbf{\color{b1color}Lingolia} - \href{https://deutsch.lingolia.com/de/grammatik}{deutsch.lingolia.com/de/grammatik}
    \item \textbf{\color{b1color}SSM Training Materialien}
\end{itemize}
